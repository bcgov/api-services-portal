import { Meta, Story, Canvas } from '@storybook/addon-docs';
import BusinessProfileContent from './business-profile-content';

<Meta title="APS/Business Profile" />

export const Template = ({ 
  businessType, 
  legalName,
  suspended,
  addressLine1,
  addressLine2,
  city,
  province,
  postal,
  ...props
}) => (
    <BusinessProfileContent
      data-testid="business-profile"
      data={{ 
        institution: {
          businessType: businessType,
          legalName: legalName,
          isSuspended: suspended,
          address: {
            addressLine1: addressLine1,
            addressLine2: addressLine2,
            city: city,
            province: province,
            postal: postal,
          },
        },
      }}
      {...props}
    />
);

# Business Profile

Displays the address information of a registered vendor.

## Loading state

`BusinessProfile` loads its own data, but the loading state is visible here.

<Canvas>
  <Story name="Loading State">
    <BusinessProfileContent isLoading />
  </Story>
</Canvas>
