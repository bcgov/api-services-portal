  Icon,
  HStack,

import { FaGripVertical, FaPlus } from 'react-icons/fa'
import { Meta, Story, Canvas } from '@storybook/addon-docs';


<Meta title="BCGov/Tags" component={Tag} />

export const Template = ({ children, ...args }) => (
  <HStack spacing={4} >
    <Tag {...args}>{children}</Tag>
    <Tag {...args}>
      <TagLabel>Tag with Close</TagLabel>

    <Tag {...args}>
      <TagLeftIcon as={FaPlus} />

    </Tag>
  </HStack>

export const DragTemplate = (args) => (
  <Tag {...args}>
    <TagLabel>Tag with Icon</TagLabel>
    <TagRightIcon as={FaGripVertical} />
  </Tag>
)

# Tags

Custom variants have been added to extend the base Chakra theme to match the BC Design Guidelines. To use:

Reference the [Chakra Button](https://chakra-ui.com/docs/form/button) documentation for additional props

```javascript
import { Tag } from '@chakra-ui/react';

const Component = () => <Tag>Tag</Tag>
```

## Default Tag

The default implementation of this tag works well for components where it is the only tag present in a card or table row.
